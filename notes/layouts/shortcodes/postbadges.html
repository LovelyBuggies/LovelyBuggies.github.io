{{/* Renders badges for Date, Reading Time, Author using Book theme styles */}} {{- $page := .Page -}} {{- $date := $page.Date -}} {{- $dateYear := $date | time.Format "2006" -}} {{- $dateFmt := cond (ne $dateYear "0001") ($date | time.Format "Jan 2, 2006") ($page.Lastmod | time.Format "Jan 2, 2006") -}} {{- $rt := or ($page.Param "readingTime") $page.ReadingTime -}} {{- $rtInt := int $rt -}} {{- $rtAdj := cond (lt $rtInt 1) 1 $rtInt -}} {{- $reading := printf "%d mins" $rtAdj -}} {{- $author := or .Site.Params.AuthorName .Site.Params.Author "Shuo" -}} Author{{ $author }} Date{{ $dateFmt }} Reading{{ $reading }} {{- /* Always show Type as Review for all posts */ -}} TypeReview
